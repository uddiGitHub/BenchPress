Q1 — Read customer by ID

Cassandra

SELECT customer_id, name, nation, region FROM customer_orders WHERE customer_id = ? LIMIT 1;

MongoDB

db.customers.find({ customer_id: ? }).limit(1)

MySQL

SELECT c.c_custkey, c.c_name, n.n_name AS nation, r.r_name AS region FROM customer c JOIN nation n ON c.c_nationkey = n.n_nationkey JOIN region r ON n.n_regionkey = r.r_regionkey WHERE c.c_custkey = ?;

Q2 — Read customer orders detail

Cassandra

SELECT * FROM customer_orders WHERE customer_id = ?;

MongoDB

db.customers.aggregate([ { $match: { customer_id: ? } }, { $unwind: { path: '$orders', preserveNullAndEmptyArrays: true } }, { $unwind: { path: '$orders.lineitems', preserveNullAndEmptyArrays: true } }, { $project: { _id: 0, customer_id: 1, name: 1, 'orders.order_id': 1 /* ... */ } } ])

MySQL

SELECT c.*, o.*, l.* FROM customer c JOIN orders o ON c.c_custkey = o.o_custkey JOIN lineitem l ON o.o_orderkey = l.l_orderkey WHERE c.c_custkey = ?;

Q3 — Aggregation: orders by status

Cassandra

// 1. Scan partition for order status SELECT order_id, line_number, status FROM customer_orders WHERE customer_id = ?; // 2. Client-side distinct count int count = 0; Set<Integer> uniqueOrders = new HashSet<>(); for (Row row : results) { uniqueOrders.add(row.getInt("order_id")); } count = uniqueOrders.size();

MongoDB

db.customers.aggregate([ { $match: { customer_id: ? } }, { $unwind: '$orders' }, { $group: { _id: '$orders.status', order_count: { $sum: 1 }, total_value: { $sum: '$orders.total_price' } }} ])

MySQL

SELECT o_orderstatus, COUNT(*) as order_count, SUM(o_totalprice) FROM orders WHERE o_custkey = ? GROUP BY o_orderstatus;

Q4 — Range / filter query

Cassandra

// Cannot do cross-partition range efficiently, // so we scan the customer's partition & filter. SELECT order_id, return_flag FROM customer_orders WHERE customer_id = ? AND return_flag = 'R' ALLOW FILTERING;

MongoDB

db.customers.find({ customer_id: { $gte: minId, $lte: maxId }, region: 'ASIA' })

MySQL

SELECT c.c_custkey, c.c_name FROM customer c JOIN nation n ON c.c_nationkey = n.n_nationkey JOIN region r ON n.n_regionkey = r.r_regionkey WHERE c.c_custkey BETWEEN ? AND ? AND r.r_name = 'ASIA';

Q5 — Revenue aggregation

Cassandra

// 1. Scan partition for line items SELECT extended_price, discount FROM customer_orders WHERE customer_id = ?; // 2. Client-side revenue sum double revenue = 0; for (Row row : results) { revenue += row.extended_price * (1 - row.discount); }

MongoDB

db.customers.aggregate([ { $match: { customer_id: ? } }, { $unwind: '$orders' }, { $unwind: '$orders.lineitems' }, { $group: { _id: null, revenue: { $sum: { $multiply: [ '$orders.lineitems.extended_price', { $subtract: [1, '$orders.lineitems.discount'] } ] } } }} ])

MySQL

SELECT SUM(l.l_extendedprice * (1 - l.l_discount)) AS revenue FROM customer c JOIN orders o ON c.c_custkey = o.o_custkey JOIN lineitem l ON o.o_orderkey = l.l_orderkey WHERE c.c_custkey = ?;
